def solution(people, limit):
    sorted(people)
    answer = 0
    first = 0
    last = len(people) - 1
    while first < last:
        if people[first] + people[last] <= limit:
            # first, last 배에 태움(if문 걸림)
            first += 1
        # last 혼자 배에 태움(if 문 안걸리면)
        last -= 1
        answer += 1
        
    # 홀수명 일 경우
    if first == last:
        answer += 1
    return answer
