import math

def solution(progresses, speeds):
    answer = []
    q = []
    for i in range(0, len(progresses)):
            q.append(math.ceil((100 - progresses[i])/speeds[i]))      
    for j in range(0, len(q)):
        count = 1
        if q[j] == 0:
            continue
        elif q[j] == q[-1]:
            answer.append(count)
            break
        for k in range(j+1, len(q)):
            if q[k] == 0:
                continue
            elif q[j] >= q[k]:
                count += 1
                q[k] = 0
            else:
                continue
        answer.append(count)
    return answer
